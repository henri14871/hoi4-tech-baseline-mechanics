##############################################
# Arms Race Mechanics — Scripted Triggers
# All triggers run in COUNTRY scope
##############################################

# Returns true if the ARM system is NOT disabled
arm_system_enabled = {
    NOT = { has_game_rule = { rule = arm_auto_research_scope option = ARM_DISABLED } }
}

# Returns true if this country should be evaluated by ARM
arm_country_eligible = {
    NOT = { has_game_rule = { rule = arm_auto_research_scope option = ARM_DISABLED } }
    has_capitulated = no
    OR = {
        num_of_civilian_factories > 0
        num_of_military_factories > 0
    }
    OR = {
        is_ai = yes
        has_game_rule = { rule = arm_auto_research_scope option = ARM_EVERYONE }
    }
}

# Checks Rule 1 = Everyone (players included)
arm_is_player_included = {
    has_game_rule = { rule = arm_auto_research_scope option = ARM_EVERYONE }
}

# At war AND war/faction catch-up rule not disabled
arm_is_at_war_combat_bonus = {
    has_war = yes
    NOT = { has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_DISABLED } }
}

# In faction with a great-power+ leader (tier > 3)
arm_is_in_faction_with_major = {
    is_in_faction = yes
    any_country = {
        is_faction_leader = yes
        is_in_faction_with = ROOT
        check_variable = { var = arm_tier_index value = 3 compare = greater_than }
    }
}

# Is subject AND puppet-sharing rule not disabled
arm_is_puppet_eligible = {
    is_subject = yes
    NOT = { has_game_rule = { rule = arm_puppet_sharing option = ARM_PUP_DISABLED } }
}

# At war, not capitulated, losing territory, rule not disabled
arm_is_desperation_eligible = {
    has_war = yes
    has_capitulated = no
    NOT = { has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_DISABLED } }
    OR = {
        surrender_progress > 0.2
        AND = {
            has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_ENHANCED }
            surrender_progress > 0.1
        }
    }
}

# Doctrine auto-research permitted for this country
arm_doctrine_allowed = {
    OR = {
        has_game_rule = { rule = arm_doctrine_handling option = ARM_DOC_EVERYONE }
        AND = {
            is_ai = yes
            has_game_rule = { rule = arm_doctrine_handling option = ARM_DOC_AI_ONLY }
        }
    }
}

# Nuclear/rockets allowed
arm_advanced_tech_allowed = {
    OR = {
        has_game_rule = { rule = arm_advanced_tech option = ARM_ADV_INCLUDED }
        AND = {
            has_game_rule = { rule = arm_advanced_tech option = ARM_ADV_SUPERPOWERS }
            check_variable = { var = arm_tier_index value = 4 compare = greater_than }
        }
    }
    check_variable = { var = arm_industry_competence value = 84 compare = greater_than }
}

# Rule 3 includes air/naval tech
arm_tech_scope_includes_air_naval = {
    OR = {
        has_game_rule = { rule = arm_tech_scope option = ARM_CORE_AIR_NAVAL }
        has_game_rule = { rule = arm_tech_scope option = ARM_ALL_EXCEPT_DOCTRINES }
        has_game_rule = { rule = arm_tech_scope option = ARM_EVERYTHING }
    }
}

# Rule 3 includes everything except doctrines
arm_tech_scope_includes_all = {
    OR = {
        has_game_rule = { rule = arm_tech_scope option = ARM_ALL_EXCEPT_DOCTRINES }
        has_game_rule = { rule = arm_tech_scope option = ARM_EVERYTHING }
    }
}

# Enemy controls at least one of our core states
arm_is_defensive_war = {
    has_war = yes
    any_enemy_country = {
        any_controlled_state = {
            is_core_of = ROOT
        }
    }
}
