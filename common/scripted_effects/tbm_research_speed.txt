# Tech Baseline Mechanics - Research Speed Spirit Swap Logic
# Scripted effect: tbm_update_research_speed_spirit
# Runs in COUNTRY scope. Called during each evaluation cycle.

tbm_clear_tier_research_spirits = {
    if = { limit = { has_idea = tbm_tier_superpower } remove_ideas = tbm_tier_superpower }
    if = { limit = { has_idea = tbm_tier_great_power } remove_ideas = tbm_tier_great_power }
    if = { limit = { has_idea = tbm_tier_regional_power } remove_ideas = tbm_tier_regional_power }
    if = { limit = { has_idea = tbm_tier_minor_industrial } remove_ideas = tbm_tier_minor_industrial }
    if = { limit = { has_idea = tbm_tier_minor } remove_ideas = tbm_tier_minor }
    if = { limit = { has_idea = tbm_tier_micro } remove_ideas = tbm_tier_micro }
}

tbm_apply_tier_research_spirit = {
    if = {
        limit = {
            check_variable = { var = tbm_tier_index value = 5 compare = equals }
            OR = {
                NOT = { has_idea = tbm_tier_superpower }
                has_idea = tbm_tier_great_power
                has_idea = tbm_tier_regional_power
                has_idea = tbm_tier_minor_industrial
                has_idea = tbm_tier_minor
                has_idea = tbm_tier_micro
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_superpower
    }
    else_if = {
        limit = {
            check_variable = { var = tbm_tier_index value = 4 compare = equals }
            OR = {
                NOT = { has_idea = tbm_tier_great_power }
                has_idea = tbm_tier_superpower
                has_idea = tbm_tier_regional_power
                has_idea = tbm_tier_minor_industrial
                has_idea = tbm_tier_minor
                has_idea = tbm_tier_micro
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_great_power
    }
    else_if = {
        limit = {
            check_variable = { var = tbm_tier_index value = 3 compare = equals }
            OR = {
                NOT = { has_idea = tbm_tier_regional_power }
                has_idea = tbm_tier_superpower
                has_idea = tbm_tier_great_power
                has_idea = tbm_tier_minor_industrial
                has_idea = tbm_tier_minor
                has_idea = tbm_tier_micro
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_regional_power
    }
    else_if = {
        limit = {
            check_variable = { var = tbm_tier_index value = 2 compare = equals }
            OR = {
                NOT = { has_idea = tbm_tier_minor_industrial }
                has_idea = tbm_tier_superpower
                has_idea = tbm_tier_great_power
                has_idea = tbm_tier_regional_power
                has_idea = tbm_tier_minor
                has_idea = tbm_tier_micro
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_minor_industrial
    }
    else_if = {
        limit = {
            check_variable = { var = tbm_tier_index value = 1 compare = equals }
            OR = {
                NOT = { has_idea = tbm_tier_minor }
                has_idea = tbm_tier_superpower
                has_idea = tbm_tier_great_power
                has_idea = tbm_tier_regional_power
                has_idea = tbm_tier_minor_industrial
                has_idea = tbm_tier_micro
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_minor
    }
    else_if = {
        limit = {
            OR = {
                NOT = { has_idea = tbm_tier_micro }
                has_idea = tbm_tier_superpower
                has_idea = tbm_tier_great_power
                has_idea = tbm_tier_regional_power
                has_idea = tbm_tier_minor_industrial
                has_idea = tbm_tier_minor
            }
        }
        tbm_clear_tier_research_spirits = yes
        add_ideas = tbm_tier_micro
    }
}

tbm_update_research_speed_spirit = {
    if = {
        limit = { has_game_rule = { rule = tbm_research_speed_bonuses option = TBM_RSB_DISABLED } }
        tbm_clear_tier_research_spirits = yes
        tbm_clear_land_research_spirits = yes
        tbm_clear_air_research_spirits = yes
        tbm_clear_naval_research_spirits = yes
        tbm_clear_industry_research_spirits = yes
    }
    else_if = {
        limit = { has_game_rule = { rule = tbm_research_speed_bonuses option = TBM_RSB_TIER_ONLY } }
        tbm_apply_tier_research_spirit = yes
        tbm_clear_land_research_spirits = yes
        tbm_clear_air_research_spirits = yes
        tbm_clear_naval_research_spirits = yes
        tbm_clear_industry_research_spirits = yes
    }
    else = {
        tbm_apply_tier_research_spirit = yes
        tbm_apply_land_research_spirit = yes
        tbm_apply_air_research_spirit = yes
        tbm_apply_naval_research_spirit = yes
        tbm_apply_industry_research_spirit = yes
    }
}
