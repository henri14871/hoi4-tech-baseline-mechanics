#---------------------------------------------------------
# Effect 1: arm_assign_power_tier
#
# Maps arm_global_power -> arm_tier_index (0..5)
# with +5 hysteresis on promotion and -5 on demotion.
# Then sets arm_base_lag and arm_quarterly_cap.
#---------------------------------------------------------
arm_assign_power_tier = {

    # --- Initialize tier if it has never been set ---
    if = {
        limit = {
            NOT = { has_country_flag = arm_tier_initialized }
        }
        set_variable = { arm_tier_index = 0 }
        set_country_flag = arm_tier_initialized
    }

    # Save old tier for change detection
    set_variable = { arm_old_tier_index = arm_tier_index }

    # ===================================================
    # PROMOTIONS (ascending — only if currently below)
    # Threshold + 5 hysteresis to promote
    # ===================================================

    # Promote 0 -> 1 (Micro -> Minor) requires > 25 (20+5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 1 compare = less_than }
            check_variable = { var = arm_global_power value = 25 compare = greater_than }
        }
        set_variable = { arm_tier_index = 1 }
    }

    # Promote 1 -> 2 (Minor -> Minor Industrial) requires > 50 (45+5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 2 compare = less_than }
            check_variable = { var = arm_global_power value = 50 compare = greater_than }
        }
        set_variable = { arm_tier_index = 2 }
    }

    # Promote 2 -> 3 (Minor Industrial -> Regional) requires > 95 (90+5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 3 compare = less_than }
            check_variable = { var = arm_global_power value = 95 compare = greater_than }
        }
        set_variable = { arm_tier_index = 3 }
    }

    # Promote 3 -> 4 (Regional -> Great Power) requires > 155 (150+5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 4 compare = less_than }
            check_variable = { var = arm_global_power value = 155 compare = greater_than }
        }
        set_variable = { arm_tier_index = 4 }
    }

    # Promote 4 -> 5 (Great Power -> Superpower) requires > 305 (300+5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 5 compare = less_than }
            check_variable = { var = arm_global_power value = 305 compare = greater_than }
        }
        set_variable = { arm_tier_index = 5 }
    }

    # ===================================================
    # DEMOTIONS (descending — only if currently above)
    # Threshold - 5 hysteresis to demote
    # ===================================================

    # Demote 5 -> 4 (Superpower -> Great Power) if score < 295 (300-5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 4 compare = greater_than }
            check_variable = { var = arm_global_power value = 295 compare = less_than }
        }
        set_variable = { arm_tier_index = 4 }
    }

    # Demote 4 -> 3 (Great Power -> Regional) if score < 145 (150-5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 3 compare = greater_than }
            check_variable = { var = arm_global_power value = 145 compare = less_than }
        }
        set_variable = { arm_tier_index = 3 }
    }

    # Demote 3 -> 2 (Regional -> Minor Industrial) if score < 85 (90-5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 2 compare = greater_than }
            check_variable = { var = arm_global_power value = 85 compare = less_than }
        }
        set_variable = { arm_tier_index = 2 }
    }

    # Demote 2 -> 1 (Minor Industrial -> Minor) if score < 40 (45-5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 1 compare = greater_than }
            check_variable = { var = arm_global_power value = 40 compare = less_than }
        }
        set_variable = { arm_tier_index = 1 }
    }

    # Demote 1 -> 0 (Minor -> Micro) if score < 15 (20-5)
    if = {
        limit = {
            check_variable = { var = arm_tier_index value = 0 compare = greater_than }
            check_variable = { var = arm_global_power value = 15 compare = less_than }
        }
        set_variable = { arm_tier_index = 0 }
    }

    # ===================================================
    # Base lag by tier
    # ===================================================
    if = {
        limit = { check_variable = { var = arm_tier_index value = 5 compare = equals } }
        set_variable = { arm_base_lag = 0.0 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 4 compare = equals } }
        set_variable = { arm_base_lag = 0.5 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 3 compare = equals } }
        set_variable = { arm_base_lag = 1.5 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 2 compare = equals } }
        set_variable = { arm_base_lag = 3.0 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 1 compare = equals } }
        set_variable = { arm_base_lag = 4.5 }
    }
    else = {
        set_variable = { arm_base_lag = 6.0 }
    }

    # ===================================================
    # Quarterly cap by tier
    # Superpower=6, Great=5, Regional=4, MinorInd=3, Minor=2, Micro=1
    # ===================================================
    if = {
        limit = { check_variable = { var = arm_tier_index value = 5 compare = equals } }
        set_variable = { arm_quarterly_cap = 6 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 4 compare = equals } }
        set_variable = { arm_quarterly_cap = 5 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 3 compare = equals } }
        set_variable = { arm_quarterly_cap = 4 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 2 compare = equals } }
        set_variable = { arm_quarterly_cap = 3 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 1 compare = equals } }
        set_variable = { arm_quarterly_cap = 2 }
    }
    else = {
        set_variable = { arm_quarterly_cap = 1 }
    }

    # ===================================================
    # Rule 7 — Grant cap modifier
    # ===================================================
    if = {
        limit = {
            has_game_rule = { rule = arm_grant_cap option = ARM_CAP_STRICT }
        }
        # Divide by 2, round up: add 1 then divide by 2 (integer)
        add_to_variable = { arm_quarterly_cap = 1 }
        divide_variable = { arm_quarterly_cap = 2 }
        # Clamp to at least 1
        clamp_variable = { var = arm_quarterly_cap min = 1 max = 999 }
    }
    else_if = {
        limit = {
            has_game_rule = { rule = arm_grant_cap option = ARM_CAP_UNLIMITED }
        }
        set_variable = { arm_quarterly_cap = 999 }
    }
    # ARM_CAP_STANDARD: use as-is (no change)

    # ===================================================
    # Rule 2 — Intensity modifier on base lag
    # ===================================================
    if = {
        limit = {
            has_game_rule = { rule = arm_auto_research_intensity option = ARM_RELAXED }
        }
        add_to_variable = { arm_base_lag = 1.0 }
    }
    else_if = {
        limit = {
            has_game_rule = { rule = arm_auto_research_intensity option = ARM_AGGRESSIVE }
        }
        subtract_from_variable = { arm_base_lag = 0.5 }
        clamp_variable = { var = arm_base_lag min = 0 max = 40 }
    }

    # ===================================================
    # Set tier average for competence bonus calculations
    # ===================================================
    if = {
        limit = { check_variable = { var = arm_tier_index value = 5 compare = equals } }
        set_variable = { arm_tier_avg = 90 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 4 compare = equals } }
        set_variable = { arm_tier_avg = 65 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 3 compare = equals } }
        set_variable = { arm_tier_avg = 40 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 2 compare = equals } }
        set_variable = { arm_tier_avg = 25 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 1 compare = equals } }
        set_variable = { arm_tier_avg = 12 }
    }
    else = {
        set_variable = { arm_tier_avg = 5 }
    }

    # --- Fire tier change event if tier changed ---
    if = {
        limit = {
            NOT = { check_variable = { var = arm_tier_index value = arm_old_tier_index compare = equals } }
            NOT = { has_game_rule = { rule = arm_notifications option = ARM_NOTIFY_SILENT } }
        }
        country_event = { id = arm_event.2 }
    }
}

    # Require Superpower (tier 5), industry >= 85
    # Also require Rule 5 (ARM_TECH_SCOPE) is ARM_EVERYTHING

    # arm_cat_nuclear — Superpower, industry >= 85, Rule 5 allows

        set_country_flag = arm_cat_nuclear
    }

    # arm_cat_rockets — Superpower, industry >= 85, Rule 5 allows
    if = {
        limit = {
            has_game_rule = { rule = arm_tech_scope option = ARM_EVERYTHING }
            check_variable = { var = arm_tier_index value = 4 compare = greater_than }
            check_variable = { var = arm_industry_competence value = 84 compare = greater_than }
        }
        set_country_flag = arm_cat_rockets
    }
}

#---------------------------------------------------------
# Effect 3: arm_compute_effective_lag
#
# Computes per-branch effective lag from arm_base_lag,
# applying competence bonuses, war bonuses, faction bonuses,
# lend-lease, desperation, and puppet overlord bonuses.
#
# Combat branches: land, air
# Tech-sharing branches: all four (industry especially
#   benefits from faction/lend-lease)
#---------------------------------------------------------
arm_compute_effective_lag = {

    # ===================================================
    # LAND branch — combat category
    # ===================================================
    set_variable = { arm_effective_land_lag = arm_base_lag }

    # --- Branch competence bonus (land) ---
    set_variable = { arm_land_comp_threshold_20 = arm_tier_avg }
    add_to_variable = { arm_land_comp_threshold_20 = 20 }
    set_variable = { arm_land_comp_threshold_40 = arm_tier_avg }
    add_to_variable = { arm_land_comp_threshold_40 = 40 }

    if = {
        limit = {
            check_variable = { var = arm_land_competence value = arm_land_comp_threshold_40 compare = greater_than }
        }
        subtract_from_variable = { arm_effective_land_lag = 1.0 }
    }
    else_if = {
        limit = {
            check_variable = { var = arm_land_competence value = arm_land_comp_threshold_20 compare = greater_than }
        }
        subtract_from_variable = { arm_effective_land_lag = 0.5 }
    }

    # --- War bonus (land is combat) ---
    # Compute war bonus and desperation bonus separately, take higher
    set_variable = { arm_land_war_bonus = 0 }
    set_variable = { arm_land_desperation_bonus = 0 }

    if = {
        limit = { has_war = yes }
        if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_ENHANCED }
            }
            set_variable = { arm_land_war_bonus = 1.0 }
        }
        else_if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_STANDARD }
            }
            set_variable = { arm_land_war_bonus = 0.5 }
        }
    }

    # --- Desperation bonus (land — combat, does not stack with war bonus) ---
    if = {
        limit = {
            has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_ENHANCED }
        }
        if = {
            limit = { surrender_progress > 0.2 }
            set_variable = { arm_land_desperation_bonus = 0.75 }
        }
        else_if = {
            limit = { surrender_progress > 0.1 }
            set_variable = { arm_land_desperation_bonus = 0.5 }
        }
    }
    else_if = {
        limit = {
            has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_STANDARD }
        }
        if = {
            limit = { surrender_progress > 0.4 }
            set_variable = { arm_land_desperation_bonus = 0.75 }
        }
        else_if = {
            limit = { surrender_progress > 0.2 }
            set_variable = { arm_land_desperation_bonus = 0.5 }
        }
    }

    # Take the higher of war bonus and desperation bonus
    if = {
        limit = {
            check_variable = { var = arm_land_desperation_bonus value = arm_land_war_bonus compare = greater_than }
        }
        subtract_from_variable = { arm_effective_land_lag = arm_land_desperation_bonus }
    }
    else = {
        subtract_from_variable = { arm_effective_land_lag = arm_land_war_bonus }
    }

    # --- Defensive war bonus (land — combat) ---
    if = {
        limit = {
            arm_is_defensive_war = yes
        }
        subtract_from_variable = { arm_effective_land_lag = 0.25 }
    }

    # --- Faction bonus (land benefits from tech sharing) ---
    if = {
        limit = {
            arm_is_in_faction_with_major = yes
        }
        if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_ENHANCED }
            }
            subtract_from_variable = { arm_effective_land_lag = 1.0 }
        }
        else_if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_STANDARD }
            }
            subtract_from_variable = { arm_effective_land_lag = 0.5 }
        }

        # --- Lend-lease enhancement (land) ---
        if = {
            limit = {
                any_country = {
                    is_lend_leasing = ROOT
                    check_variable = { var = arm_tier_index value = 3 compare = greater_than }
                }
            }
            subtract_from_variable = { arm_effective_land_lag = 0.25 }
        }
    }

    # --- Puppet bonus (land) ---
    if = {
        limit = {
            arm_is_puppet_eligible = yes
            overlord = {
                check_variable = { var = arm_tier_index value = 4 compare = greater_than }
            }
        }
        subtract_from_variable = { arm_effective_land_lag = 1.0 }
    }
    else_if = {
        limit = {
            arm_is_puppet_eligible = yes
            overlord = {
                check_variable = { var = arm_tier_index value = 3 compare = greater_than }
            }
        }
        subtract_from_variable = { arm_effective_land_lag = 0.5 }
    }

    # Clamp land lag
    clamp_variable = { var = arm_effective_land_lag min = 0 max = 40 }

    # ===================================================
    # AIR branch — combat category
    # ===================================================
    set_variable = { arm_effective_air_lag = arm_base_lag }

    # --- Branch competence bonus (air) ---
    set_variable = { arm_air_comp_threshold_20 = arm_tier_avg }
    add_to_variable = { arm_air_comp_threshold_20 = 20 }
    set_variable = { arm_air_comp_threshold_40 = arm_tier_avg }
    add_to_variable = { arm_air_comp_threshold_40 = 40 }

    if = {
        limit = {
            check_variable = { var = arm_air_competence value = arm_air_comp_threshold_40 compare = greater_than }
        }
        subtract_from_variable = { arm_effective_air_lag = 1.0 }
    }
    else_if = {
        limit = {
            check_variable = { var = arm_air_competence value = arm_air_comp_threshold_20 compare = greater_than }
        }
        subtract_from_variable = { arm_effective_air_lag = 0.5 }
    }

    # --- War bonus (air is combat) ---
    set_variable = { arm_air_war_bonus = 0 }
    set_variable = { arm_air_desperation_bonus = 0 }

    if = {
        limit = { has_war = yes }
        if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_ENHANCED }
            }
            set_variable = { arm_air_war_bonus = 1.0 }
        }
        else_if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_STANDARD }
            }
            set_variable = { arm_air_war_bonus = 0.5 }
        }
    }

    # --- Desperation bonus (air — combat) ---
    if = {
        limit = {
            has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_ENHANCED }
        }
        if = {
            limit = { surrender_progress > 0.2 }
            set_variable = { arm_air_desperation_bonus = 0.75 }
        }
        else_if = {
            limit = { surrender_progress > 0.1 }
            set_variable = { arm_air_desperation_bonus = 0.5 }
        }
    }
    else_if = {
        limit = {
            has_game_rule = { rule = arm_anti_death_spiral option = ARM_ADS_STANDARD }
        }
        if = {
            limit = { surrender_progress > 0.4 }
            set_variable = { arm_air_desperation_bonus = 0.75 }
        }
        else_if = {
            limit = { surrender_progress > 0.2 }
            set_variable = { arm_air_desperation_bonus = 0.5 }
        }
    }

    # Take the higher of war bonus and desperation bonus
    if = {
        limit = {
            check_variable = { var = arm_air_desperation_bonus value = arm_air_war_bonus compare = greater_than }
        }
        subtract_from_variable = { arm_effective_air_lag = arm_air_desperation_bonus }
    }
    else = {
        subtract_from_variable = { arm_effective_air_lag = arm_air_war_bonus }
    }

    # --- Defensive war bonus (air — combat) ---
    if = {
        limit = {
            arm_is_defensive_war = yes
        }
        subtract_from_variable = { arm_effective_air_lag = 0.25 }
    }

    # --- Faction bonus (air) ---
    if = {
        limit = {
            arm_is_in_faction_with_major = yes
        }
        if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_ENHANCED }
            }
            subtract_from_variable = { arm_effective_air_lag = 1.0 }
        }
        else_if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_STANDARD }
            }
            subtract_from_variable = { arm_effective_air_lag = 0.5 }
        }

        # --- Lend-lease enhancement (air) ---
        if = {
            limit = {
                any_country = {
                    is_lend_leasing = ROOT
                    check_variable = { var = arm_tier_index value = 3 compare = greater_than }
                }
            }
            subtract_from_variable = { arm_effective_air_lag = 0.25 }
        }
    }

    # --- Puppet bonus (air) ---
    if = {
        limit = {
            arm_is_puppet_eligible = yes
            overlord = {
                check_variable = { var = arm_tier_index value = 4 compare = greater_than }
            }
        }
        subtract_from_variable = { arm_effective_air_lag = 1.0 }
    }
    else_if = {
        limit = {
            arm_is_puppet_eligible = yes
            overlord = {
                check_variable = { var = arm_tier_index value = 3 compare = greater_than }
            }
        }
        subtract_from_variable = { arm_effective_air_lag = 0.5 }
    }

    # Clamp air lag
    clamp_variable = { var = arm_effective_air_lag min = 0 max = 40 }

    # ===================================================
    # NAVAL branch — NOT a combat category for war/desperation
    # but IS a tech-sharing (faction) category
    # ===================================================
    set_variable = { arm_effective_naval_lag = arm_base_lag }

    # --- Branch competence bonus (naval) ---
    set_variable = { arm_naval_comp_threshold_20 = arm_tier_avg }
    add_to_variable = { arm_naval_comp_threshold_20 = 20 }
    set_variable = { arm_naval_comp_threshold_40 = arm_tier_avg }
    add_to_variable = { arm_naval_comp_threshold_40 = 40 }

    if = {
        limit = {
            check_variable = { var = arm_naval_competence value = arm_naval_comp_threshold_40 compare = greater_than }
        }
        subtract_from_variable = { arm_effective_naval_lag = 1.0 }
    }
    else_if = {
        limit = {

        }
        subtract_from_variable = { arm_effective_naval_lag = 0.5 }
    }

    # Naval does NOT get war bonus or desperation bonus (not combat)

    # --- Faction bonus (naval) ---
    if = {

        if = {

            }
            subtract_from_variable = { arm_effective_naval_lag = 1.0 }
        }
        else_if = {
            limit = {
                has_game_rule = { rule = arm_war_faction_catchup option = ARM_WF_STANDARD }
            }
            subtract_from_variable = { arm_effective_naval_lag = 0.5 }
        }

        # --- Lend-lease enhancement (naval) ---
        if = {
            limit = {
                any_country = {
                    is_lend_leasing = ROOT
                    check_variable = { var = arm_tier_index value = 3 compare = greater_than }

            }
            subtract_from_variable = { arm_effective_naval_lag = 0.25 }

    }

    # --- Puppet bonus (naval) ---
    if = {

            overlord = {

    # Industry does NOT get war bonus or desperation bonus (not combat)

    if = {

        }

        # --- Lend-lease enhancement (industry) ---
        if = {

                    check_variable = { var = arm_tier_index value = 3 compare = greater_than }
                }
            }
            subtract_from_variable = { arm_effective_industry_lag = 0.25 }

#---------------------------------------------------------
# Effect 4: arm_compute_target_years
#
# Subtracts each branch's effective lag from the current
# game year to produce the target tech year per branch.
#---------------------------------------------------------
arm_compute_target_years = {
    set_variable = { var = arm_target_year_land value = global.arm_current_year }
    subtract_from_variable = { arm_target_year_land = arm_effective_land_lag }

    set_variable = { var = arm_target_year_air value = global.arm_current_year }
    subtract_from_variable = { arm_target_year_air = arm_effective_air_lag }

    set_variable = { var = arm_target_year_naval value = global.arm_current_year }
    subtract_from_variable = { arm_target_year_naval = arm_effective_naval_lag }

    set_variable = { var = arm_target_year_industry value = global.arm_current_year }
    subtract_from_variable = { arm_target_year_industry = arm_effective_industry_lag }
}

#---------------------------------------------------------
# Effect 5: arm_compute_quarterly_cap
#
# Convenience alias — the quarterly cap is already computed
# inside arm_assign_power_tier. This effect can be called
# independently if only the cap needs recalculation
# (e.g. after a mid-quarter rule change).
#---------------------------------------------------------
arm_compute_quarterly_cap = {
    # Recompute cap from current tier
    if = {
        limit = { check_variable = { var = arm_tier_index value = 5 compare = equals } }
        set_variable = { arm_quarterly_cap = 6 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 4 compare = equals } }
        set_variable = { arm_quarterly_cap = 5 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 3 compare = equals } }
        set_variable = { arm_quarterly_cap = 4 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 2 compare = equals } }
        set_variable = { arm_quarterly_cap = 3 }
    }
    else_if = {
        limit = { check_variable = { var = arm_tier_index value = 1 compare = equals } }
        set_variable = { arm_quarterly_cap = 2 }
    }
    else = {
        set_variable = { arm_quarterly_cap = 1 }
    }

    # Apply Rule 7 — Grant cap modifier
    if = {
        limit = {
            has_game_rule = { rule = arm_grant_cap option = ARM_CAP_STRICT }
        }
        add_to_variable = { arm_quarterly_cap = 1 }
        divide_variable = { arm_quarterly_cap = 2 }
        clamp_variable = { var = arm_quarterly_cap min = 1 max = 999 }
    }
    else_if = {
        limit = {
            has_game_rule = { rule = arm_grant_cap option = ARM_CAP_UNLIMITED }
        }
        set_variable = { arm_quarterly_cap = 999 }
    }
}
