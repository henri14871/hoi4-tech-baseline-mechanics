###############################################################################
# tbm_compat_runtime.txt
# Core compatibility runtime wrappers.
#
# Optional compatibility bundles are generated under compat_generated/ and are
# intended to load after TBM and the target overhaul mod. Those bundles replace
# the generic tbm_assign_power_tier / tbm_grant_* effects directly, so the core
# mod only needs stable wrapper hooks plus auto-detection flags.
###############################################################################

tbm_initialize_doctrine_paths_for_profile = {
    if = {
        limit = {
            tbm_doctrine_allowed = yes
            OR = {
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_VANILLA }
                AND = {
                    has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_AUTO }
                    OR = {
                        NOT = { has_global_flag = tbm_compat_auto_profile_detected }
                        has_global_flag = tbm_compat_auto_doctrine_safe
                    }
                }
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_KAISERREICH }
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_ENDSIEG }
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_ETT1960 }
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_KAISERREDUX }
                has_game_rule = { rule = tbm_compat_profile option = TBM_COMPAT_ROAD_TO_56 }
            }
        }
        tbm_initialize_doctrine_paths = yes
    }
}
